\chapter{Stochastic Gradient Methods}
\label{Ch:SGD}

\section{SGD}

Stochastic Gradient Descent (SGD) is a stochastic analogue of the Gradient Descent algorithm, aiming at finding
the local or global minimizers of the function expectation parameterized by some random variable.
Schematically, the algorithm can be interpreted as targeting at finding a local minimum point of
the expectation of function
\begin{equation}\label{Eq:OptimizationObjective}
f(x)=\mathbf{E} f(x;\zeta) \ ,
\end{equation}
where the index random variable $\zeta$ follows some
prescribed distribution $\mathcal{D}$, and the weight vector $x\in \mathbb{R}^d$.
SGD updates via the iteration
\begin{equation}\label{Eq:StochasticGradientDescent}
x^{k+1}=x^k-\al_k g(x^k; \zeta_k) \ ,
\end{equation}
where the noisy gradient $g(x; \zt)=\grad f(x; \zt)$, $\al_k>0$ is the learning rate, and $\zeta_k$ are i.i.d. random variables
that have the same distribution as $\zeta\sim \cD$.

In particular, in the case of supervised learning, assume the input training data is given by $\{(a_i, b_i)\}_{i=1}^n$ and the fitting function is $g(a)=b$. Then the loss function is $\ell_{(a,b)}(x)=L(g(a), b; x)$ where $x$ is the model weight. Let $f_i(x)=\ell_{(a_i, b_i)}(x)$. Then the optimization problem has the objective function
$$
f(x)=\frac{1}{n}\sum_{i=1}^n f_i(x) \ .
$$
In this case, if we run GD or AGD for $f$ directly, we have  
$$\begin{array}{ll}
\text{GD:}
& \qquad
x^{k+1} = x^{k} - \eta \nabla f\!\left(x^{k}\right) \ ,
\\
\text{Nesterov AGD:}
& \qquad
x^{k+1}
=
x^{k}
-
\alpha \nabla f\!\left(x^k+\beta\bigl(x^k-x^{k-1}\bigr)\right)
+
\beta\bigl(x^k-x^{k-1}\bigr) \ .
\end{array}$$

This means at each iteration step we have to compute $\play{\nabla f(x)
= \dfrac{1}{n}\sum_{i=1}^n \nabla f_i(x)}$. 
Usually $n$ is large therefore gradient queries of $\nabla f_i$ is large for each iteration step: if we run $T$ the iterations, the total computational cost is $O(T\times n)$. Assume $T=1000$, $n=1000$, then $T\times n = 10^6 \gg 1$. To mitigate this, we use SGD where the random variable $\zeta$ samples size $m$ ($m\leq n$) mini--batches $\mathcal{B}$ uniformly from an index set $\{1,2,...,n\}$:
$\mathcal{B}\subset \{1,2,...,n\}$ and $|\mathcal{B}|=m$ (batchsize). In this case, given loss functions $f_1(x), ..., f_n(x)$ on
training data, we have
$$f(x; \zeta)=\dfrac{1}{m}\sum\limits_{i\in \mathcal{B}}f_i(x) \text{ and } f(x)=\dfrac{1}{n}\sum\limits_{i=1}^n f_i(x) \ .$$

\begin{algorithm}[H]
\caption{Stochastic Gradient Descent}
\label{Alg:SGD}
\begin{algorithmic}[1]
\STATE \textbf{Input}: Input initialization $x^0$, index random variable $\zt\sim \cD$, noisy gradient $g(x;\zt)=\grad f(x;\zt)$ such that $\E g(x;\zt)=\grad f(x)$, learning rate $\al_k>0$, $k=0,1,2,...$
\FOR {$k=0,1,2,...,K-1$}
    \STATE Sample the i.i.d. random variable $\zt_k\sim \cD$
    \STATE Calculate the noisy gradient $g(x^k; \zt_k)=\grad f(x^k; \zt_k)$
    \STATE $x^{k+1}=x^k-\al_k g(x^k; \zeta_k)$
\ENDFOR
\STATE \textbf{Output}: Output $x^K$
\end{algorithmic}
\end{algorithm}


\begin{example}
\emph{Assume $m=1$, then $\cB=\{i_k\}$ where $i_k$ is drawn uniformly randomly from $\{1,...,n\}$ independent of $k$. Then we compare
$$\begin{array}{lll}
\text{GD}: &
x^{k+1}
=
x^k
-
\dfrac{\eta}{n}\sum\li_{i=1}^n \nabla f_i(x^k) \ ,
& 
\text{ complexity } O(T\times n) \ ,
\\
\text{SGD}: &
x^{k+1}=x^k-\eta \nabla f_{i_k}(x^k) \ ,
& \text{ complexity } O(T) \ .
\end{array}$$
Only one query of $\nabla f_i$ at each iteration.}
\end{example}

\begin{lemma}\label{Lm:MinibatchExpectationUnbiased}
We have, for any $f_i(x)$, $i=1,2,...,n$, that
$$\E\left(\dfrac{1}{m}\sum\li_{i\in \cB}f_i(x)\right)=\dfrac{1}{n}\sum\li_{i=1}^n f_i(x) \ .$$
\end{lemma}

\begin{proof}
$B_k$ are a sequence of i.i.d. mini-batches: $B_k \subset \{1,\ldots,n\}
\ , \ 
|B_k|=m$. Since $B_k$ is taken uniformly randomly among all size $m$-subsets of $\{1,\ldots,n\}$, the total number of size-$m$ subsets is $$\binom{n}{m}=\frac{n!}{m!(n-m)!} \ .$$ 
When we enumerate all ${n\choose m}$ mini-batches $\cB$, each index $i\in \{1,...,n\}$ will be repeated $n-1 \choose m-1$ times. 
This gives
$$\begin{array}{ll}
\E
\left(
\dfrac{1}{m}\sum\li_{i\in B_k} f_i(x)
\right)
& =
\dfrac{
\dfrac{1}{m}\sum\li_{i=1}^n \binom{n-1}{m-1} f_i(x)
}{
\binom{n}{m}
}
\\
& =
\dfrac{1}{m}
\cdot
\dfrac{(n-1)!}{(m-1)!(n-m)!}
\cdot
\dfrac{1}{\binom{n}{m}}
\sum\li_{i=1}^n f_i(x)
=
\dfrac{1}{n}\sum\li_{i=1}^n f_i(x) \ .
\end{array}$$
\end{proof}


Let us analyze the convergence properties of SGD. To measure the rate of convergence, our metrics are set as follows:

\begin{itemize}
\item $f$ is strongly convex: the metric is $\E\left[\|x^k-x^*\|^2\right]$.

\item $f$ is convex but not necessarily strongly convex: the metric is 
$f(x^k)-f(x^*)$.
\end{itemize}

We make the following assumption regarding the random gradient surrogate $g(x; \zt)$ used in (\ref{Eq:StochasticGradientDescent}):

\begin{assumption}\label{Assumption:SGDBasicAssumption}
We assume that for some $L_g>0$ and $B>0$ we have
\begin{equation}\label{Assumption:SGDBasicAssumption:Eq:Assumption}
\E_{\zt}\left[\|g(x;\zt)\|_2^2\right]\leq L_g^2\|x-x^*\|^2+B^2
\end{equation}
for all $x$.
\end{assumption}

\begin{remark}\label{Remark:SGD-Assumption}
In the mini-batch stochastic gradient case, $g(x;\zt)=\dfrac{1}{m}\sum\li_{i\in \cB}\grad f_i(x)$. Then we have
$$\begin{array}{ll}
&\E_\zt\|g(x;\zt)\|_2^2
\\
= & \E\left\|\dfrac{1}{m}\sum\li_{i\in \cB}\grad f_i(x)\right\|^2
\\
= & \E\left\|\dfrac{1}{m}\sum\li_{i\in \cB}(\grad f_i(x)-\grad f_i(x^*))+\dfrac{1}{m}\sum\li_{i\in \cB}\grad f_i(x^*)\right\|^2
\\
\leq & 2\E\left\|\dfrac{1}{m}\sum\li_{i\in \cB}(\grad f_i(x)-\grad f_i(x^*))\right\|^2+2\E\left\|\dfrac{1}{m}\sum\li_{i\in \cB}\grad f_i(x^*)\right\|^2
\\
\leq & 2\E\sum\li_{i\in \cB}\dfrac{1}{m^2}\sum\li_{i\in \cB}\|\grad f_i(x)-\grad f_i(x^*)\|^2+B^2
\\
\leq & 2\E\dfrac{1}{m}\sum\li_{i\in \cB}\|\grad f_i(x)-\grad f_i(x^*)\|^2+B^2
\\
\leq & 2\E\|\grad f_i(x)-\grad f_i(x^*)\|^2+B^2 \ .
\end{array}$$
So we can pick $L_g=\sqrt{2}L_{\grad f}$. If $L_g=0$ in \eqref{Assumption:SGDBasicAssumption:Eq:Assumption}, then $f$ cannot be $m$-strongly convex over an unbounded domain, since by Jensen's inequality 
$$\|\grad f(x)\|^2=\|\E g(x; \zt)\|^2\leq \E\left[\|g(x;\zt)\|^2\right]\stackrel{\text{since } L_g=0}{\leq} B^2$$ but 
$$\begin{array}{ll}
& \|x-x^*\|\cdot \|\grad f(x)\|
\\
\geq & \langle x-x^*, \grad f(x)\rangle
\\
= & \langle x-x^*, \grad f(x)-\grad f(x^*)\rangle
\\
= & \left\langle x-x^*, \play{\int_0^1} \grad^2 f(x^*+\gm(x-x^*))(x-x^*)d\gm\right\rangle \geq m\|x-x^*\|^2
\end{array}$$ for all $x$. This means $\|\grad f(x)\|^2\geq m^2\|x-x^*\|^2$, which further implies $B\geq m\|x-x^*\|$, a contradiction.
\end{remark}


\begin{theorem}\label{Thm:ConvergenceSGD} Assume $B>0$ and $L_g>0$, and $f$ is $m$-strongly convex with unique minimizer $x^*$. We choose $\al\in \left(0, \dfrac{2m}{L_g^2}\right)$ such that $1-2m\al+\al^2L_g^2>0$. For the iteration \emph{(\ref{Eq:StochasticGradientDescent})} we set $\al_k=\al$. Then
\begin{equation}\label{Thm:ConvergenceSGD:Eq:ConvergenceStronglyConvex}
\E\|x^k-x^*\|^2\leq \left(1-2m\al+\al^2L_g^2\right)^k\|x^0-x^*\|^2+\dfrac{\al^2 B^2}{2m-\al L_g^2} \ .
\end{equation}
\end{theorem}

\begin{proof}
We perform a square expansion for the stochastic iteration (\ref{Eq:StochasticGradientDescent}) to get

\begin{equation}\label{SquareExpansion:SGD}
\begin{array}{ll}
\|x^{k+1}-x^*\|^2&=\|x^k-\al_k g(x^k; \zt_k)-x^*\|^2
\\
&=\|x^k-x^*\|^2-2\al_k\langle g(x^k; \zt_k), x^k-x^*\rangle+\al_k^2\|g(x^k; \zt_k)\|^2 \ .
\end{array}
\end{equation}

So
$$\E\,\|x^{k+1}-x^*\|^2
=
\E\,\|x^k-x^*\|^2
-2\alpha_k\,\E\,\bigl\langle g(x^k;\zt_k),\,x^k-x^*\bigr\rangle
+\alpha_k^2\,\E\,\|g(x^k;\zt_k)\|^2 .
$$

We have
$$\begin{array}{ll}
& \E\,\bigl[\langle g(x^k;\zt_k),\,x^k-x^*\rangle\bigr]
\\
= &
\E\!\left[
\E_{\zt_k}
\Bigl[
\bigl\langle g(x^k;\zt_k),\,x^k-x^*\bigr\rangle
\,\Big|\,\zt_0,\ldots,\zt_{k-1}
\Bigr]
\right]
\qquad \text{(``telescoping'')}
\\
= & 
\E\!\left[
\left\langle
\E_{\zt_k}\!\Bigl[g(x^k;\zt_k)\,\big|\,\zt_0,\ldots,\zt_{k-1}\Bigr],
\,x^k-x^*
\right\rangle
\right]
\\
= & 
\E\,\bigl[\langle \nabla f(x^k),\,x^k-x^*\rangle\bigr] \ .
\end{array}
$$
Here in the last step we have used the fact that $
\E_{\zt}\,g(x;\zt)=\nabla f(x)$. Similarly
$$
\E\,\|g(x^k;\zt_k)\|^2
=
\E\!\left[
\E_{\zt_k}
\Bigl[
\|g(x^k;\zt_k)\|_2^2
\,\Big|\,\zt_0,\ldots,\zt_{k-1}
\Bigr]
\right]
\le
\E\!\left[
L_g^2\|x^k-x^*\|^2 + B^2
\right].
$$

Set $a_k=\E\left[\|x^k-x^*\|^2\right]$, then from the above and (\ref{SquareExpansion:SGD}) we get

\begin{equation}\label{SquareExpansionIteration:SGD}
a_{k+1}\leq \left(1+\al_k^2L_g^2\right)a_k-2\al_k\E\left[\langle\grad f(x^k), x^k-x^*\rangle\right]+\al_k^2B^2 \ .
\end{equation}

If $B>0$, $L_g>0$ and $f$ has strong-convexity constant $m>0$, then $
\nabla^2 f(x) \succeq m I$ and the unique minimizer $x^*$ satisfies $\grad f(x^*)=0$. Since
$$
\langle \nabla f(x)-\nabla f(x^*), x-x^*\rangle
=
\int_0^1 \langle \nabla^2 f\bigl(x^*+\gamma(x-x^*)\bigr)(x-x^*), x-x^*\rangle \, d\gamma \geq m \|x-x^*\|^2 \ ,
$$
we have $
\bigl\langle \nabla f(x),\,x-x^*\bigr\rangle
\ge m \|x-x^*\|^2$. Putting $x=x^k$ and taking $\E$-expectation we get
$\E\bigl[\langle \nabla f(x^k),\,x^k-x^*\rangle\bigr]
\ge m\,\E\|x^k-x^*\|^2$. This together with (\ref{SquareExpansionIteration:SGD}) gives 
$$a_{k+1}\leq (1-2m\al_k +\al_k^2 L_g^2)a_k+\al_k^2B^2 \ .$$

Set $\al_k=\al$ and $\mu = 1-2m\al+\al^2L_g^2$, then the above gives
$$a_{k+1}\leq \mu^{k+1}a_0+\al^2B^2(1+\mu+...+\mu^k)=\mu^{k+1}a_0+\dfrac{\al^2B^2(1-\mu^k)}{1-\mu} \ .$$
When $\mu \in (0,1)$ this gives (\ref{Thm:ConvergenceSGD:Eq:ConvergenceStronglyConvex}).
\end{proof}

We see from (\ref{Thm:ConvergenceSGD:Eq:ConvergenceStronglyConvex}) that there is a ``threshold value" $\dfrac{\al^2 B^2}{2m-\al L_g^2}$ which prevents SGD to converge (i.e. there is always a fluctuation). To over comes this, ``epoch doubling" and ``hyper-parameter tuning" have been used. 

\begin{remark}\label{Remark:ConvergenceSGD:OtherCases}
Assume $f$ is convex. We can similarly discuss the case when $L_g=0$ or $B=0$:
\begin{itemize}
\item[\emph{(1)}] Assume $L_g=0$. Let the average output $\bar{x}^K=\dfrac{\sum\li_{j=0}^K \al_jx^j}{\sum\li_{j=0}^K \al_j}$. Set $D_0^2=\|x^0-x^*\|^2$,  $\al_k=\al>0$ and $\al_{\text{opt}}=\dfrac{D_0}{B\sqrt{T+1}}$, $\tht=\dfrac{\al}{\al_{\text{opt}}}$. Then
$$\E\left[f(\bar{x}^T)-f(x^*)\right]\leq \left(\dfrac{1}{2}\tht+\dfrac{1}{2}\tht^{-1}\right)\dfrac{BD_0}{\sqrt{T+1}} \ .$$

\item[\emph{(2)}] Assume $B=0$ and $\langle\grad f(x), x-x^*\rangle\geq m\|x-x^*\|^2$ for some $m>0$. If $\al_k=\al\in \left(0, \dfrac{2m}{L_g^2}\right)$, 
then we obtain a linear rate of convergence. Optimal $\al=\dfrac{2}{L_g^2}$ and
in this case $$a_k\leq \left(1-\dfrac{m^2}{L_g^2}\right)^kD_0^2< \ve \Longleftrightarrow T\geq \left\lceil 
\dfrac{L_g^2}{m^2}\ln\left(\dfrac{D_0^2}{\ve}\right)\right\rceil \ .$$
\end{itemize}
\end{remark}

\begin{proof}[Proof of Remark 5.4]
Since $f$ is convex, by (\ref{Thm:ConvexFunctionMainTheorem:Eq:FirstOrder}) we have $$f(y)-f(x)\geq \langle \grad f(x), y-x \rangle \ .$$ 
Then the basic one-step estimate (\ref{SquareExpansionIteration:SGD}) can be written as
\begin{equation}\label{Remark:ConvergenceSGD:OtherCases:Eq:basic-recursion}
a_{k+1}
\le
a_k
-2\alpha_k\,\E\bigl[f(x^k)-f(x^*)\bigr]
+\alpha_k^2\Bigl(B^2+L_g^2 a_k\Bigr),
\end{equation}
where $a_k=\E \|x^k-x^*\|^2$.

We now discuss the two special cases.

\noindent{(1) $L_g=0$.} In this case (\ref{Remark:ConvergenceSGD:OtherCases:Eq:basic-recursion}) reduces to
\[
a_{k+1}
\le
a_k
-2\alpha_k\,\E \bigl[f(x^k)-f(x^*)\bigr]
+\alpha_k^2 B^2.
\]
Hence
\[
2\alpha_k\,\E \bigl[f(x^k)-f(x^*)\bigr]
\le
a_k-a_{k+1}+\alpha_k^2 B^2.
\]
Summing from $k=0$ to $T$ gives
\[
2\sum_{k=0}^T \alpha_k\,\E \bigl[f(x^k)-f(x^*)\bigr]
\le
a_0-a_{T+1}+B^2\sum_{k=0}^T \alpha_k^2
\le
D_0^2+B^2\sum_{k=0}^T \alpha_k^2 \ ,
\]
where $D_0^2=a_0=\|x^0-x^*\|^2$. 
Let $\bar x^K:=\dfrac{\sum\li_{j=0}^K \alpha_j x^j}{\sum\li_{j=0}^K \alpha_j}$. By convexity of $f$,
$
f(\bar x^T)
\le
\dfrac{\sum\li_{k=0}^T \alpha_k f(x^k)}{\sum\li_{k=0}^T \alpha_k}$. Taking expectation and using the previous inequality, we obtain
\[
\E \bigl[f(\bar x^T)-f(x^*)\bigr]
\le
\dfrac{D_0^2+B^2\sum\li_{k=0}^T \alpha_k^2}{2\sum\li_{k=0}^T \alpha_k} \ .
\]

Now choose $\alpha_k=\alpha>0$ for all $k$. Then $\sum\li_{k=0}^T \alpha_k=(T+1)\alpha$,
$\sum\li_{k=0}^T \alpha_k^2=(T+1)\alpha^2$,
so
$$
\E\bigl[f(\bar x^T)-f(x^*)\bigr]
\le
\frac{D_0^2}{2(T+1)\alpha}+\frac{\alpha B^2}{2} \ .$$
Minimizing the right-hand side of the above leads to $$\E \bigl[f(\bar x^T)-f(x^*)\bigr]
\le
\left(\frac{1}{2}\theta+\frac{1}{2}\theta^{-1}\right)
\frac{BD_0}{\sqrt{T+1}} \ .$$

\noindent{(2) $B=0$.}
Since we have assumed $\langle \nabla f(x),x-x^*\rangle \ge m\|x-x^*\|^2$ for some $m>0$, the basic one-step estimate (\ref{SquareExpansionIteration:SGD}) can be written as
$$a_{k+1}
\le
a_k-2m\alpha_k a_k+\alpha_k^2 L_g^2 a_k
=
\bigl(1-2m\alpha_k+L_g^2\alpha_k^2\bigr)a_k \ .$$

Now choose $\alpha_k=\alpha$ constant. Then
\[
a_{k+1}\le \bigl(1-2m\alpha+L_g^2\alpha^2\bigr)a_k.
\]
If
$0<\alpha<\dfrac{2m}{L_g^2}$,
then
$0<1-2m\alpha+L_g^2\alpha^2<1$,
so the method converges linearly as
$$
a_k
\le
\bigl(1-2m\alpha+L_g^2\alpha^2\bigr)^k D_0^2 \ .$$

Minimizing the contraction factor $1-2m\alpha+L_g^2\alpha^2$ leads to the optimal $
\alpha_{\mathrm{opt}}=\dfrac{m}{L_g^2}$.
Therefore
$a_k
\le
\left(1-\dfrac{m^2}{L_g^2}\right)^k D_0^2$.
\end{proof}


\section{Stochastic Variance-Reduced Gradient Descent (SVRG) and its variants}

\subsection{SVRG and its Convergence Proof}

As we have seen in Theorem \ref{Thm:ConvergenceSGD}, the ``threshold value" $\dfrac{\al^2 B^2}{2m - \al L_g^2}$ prevents SGD to converge. This is due to the ``variance term" $B>0$. This inspires Variance Reduction, whose goal is to reduce the variance produced by the stochastic terms in the iteration scheme.

We will introduce in detail the Stochastic Variance Reduced Gradient (SVRG) method (see \cite{johnsonzhangsvrg2013}) and provide a detailed analysis of its convergence.

Goal is to minimize $\min P(w)$ where $P(w)=\dfrac{1}{n}\sum\li_{i=1}^n \psi_i(w)$.

SGD: draw $i_t$ randomly from $\{1,...,n\}$ and set
$$w^{(t)}=w^{(t-1)}-\eta_t \grad \psi_{i_t}(w^{(t-1)}) \ .$$

How to reduce variance? We can keep a snapshot $\widetilde{w}$ after every $m$ SGD iterations, and we
maintain the \textit{average gradient}
$$\widetilde{\mu}=\grad P(\widetilde{w})=\dfrac{1}{n}\sum\li_{i=1}^n \grad \psi_i(\widetilde{w}) \ .$$

This leads to the variance--reduction scheme as follows:
$$w^{(t)}=w^{(t-1)}-\eta_t\left(\grad \psi_{i_t}(w^{(t-1)})-\grad \psi_{i_t}(\widetilde{w})+\widetilde{\mu}\right) \ .$$

\begin{algorithm}[H]
\caption{Stochastic Variance Reduced Gradient (SVRG)}
\label{Alg:SVRG}
\begin{algorithmic}[1]
\STATE \textbf{Input}: Input initialization $w^{(0)}$, update frequency $m$ and learning rate $\eta$
\FOR {$s=1,2,...$}
    \STATE $\widetilde{w}=\widetilde{w}_{s-1}$
    \STATE $\widetilde{\mu}=\dfrac{1}{n}\sum\li_{i=1}^n \grad \psi_i(\widetilde{w})$
    \STATE $w^{(0)}=\widetilde{w}$
    \FOR {$t=1,2,...,m$}
        \STATE Sample $i_t$ independently and uniformly from $\{1,...,n\}$
        \STATE Update the weight $w^{(t)}=w^{(t-1)}-\eta \left(\grad \psi_{i_t}(w^{(t-1)})-\grad \psi_{i_t}(\widetilde{w})+\widetilde{\mu}\right)$
    \ENDFOR
    \STATE $\widetilde{w}_s\leftarrow w^{(t)}$ for randomly chosen $t\in \{0,...,m-1\}$
\ENDFOR
\STATE \textbf{Output}: $\widetilde{w}_s$
\end{algorithmic}
\end{algorithm}

Suppose the SVRG Algorithm \ref{Alg:SVRG} indeed converge, then $\grad \psi_{i_t}(w^{(t-1)})-\grad \psi_{i_t}(\widetilde{w})\ra 0$ and $\widetilde{\mu}\ra 0$, leading to the fact that 
$$\text{Var}\left(\grad \psi_{i_t}(w^{(t-1)})-\grad \psi_{i_t}(\widetilde{w})+\widetilde{\mu}\right)\ra 0 \ ,$$
while
$$\E\left(\grad \psi_{i_t}(w^{(t-1)})-\grad \psi_{i_t}(\widetilde{w})+\widetilde{\mu}\right)\ra \grad P(w^{(t-1)}) \ .$$
That is to say, we have reached variance reduction. Figure \ref{Fig:SVRG-epochs} shows the process.

Here we provide a rigorous proof of the convergence of SVRG method by following the original SVRG paper.

\begin{theorem}\label{Thm:ConvergenceSVRG}
Consider the \emph{SVRG Algorithm \ref{Alg:SVRG}} and assume that each $\psi_i$ is convex, each $\grad \psi_i$ is $L$-Lipschitz, and the function
$P(w)=\dfrac{1}{n}\sum\li_{i=1}^n \psi_i(w)$ is $\gm$--strongly convex. Set $w^*=\arg\min\li_{w}P(w)$
and assume that the epoch length $m$ is so large that
\begin{equation}\label{Thm:ConvergenceSVRG:Eq:AlphaCondition}
\al=\dfrac{1}{\gm\eta(1-2L\eta)m}+\dfrac{2L\eta}{1-2L\eta}<1 \ ,
\end{equation}
then we have the \emph{linear convergence in expectation of SVRG Algorithm \ref{Alg:SVRG}}, i.e.
\begin{equation}\label{Thm:ConvergenceSVRG:Eq:LinearConvergence}
\E [P(\widetilde{w}_s)- P(w^*)]\leq \al^s [P(\widetilde{w}_0)-P(w^*)] \ .
\end{equation}
\end{theorem}

\begin{remark}\label{Rmk:Thm:ConvergenceSVRG:ChoiceParameters}
\rm We can pick $\al=\dfrac{1}{2}$, for which we just need $\dfrac{1}{\gm \eta (1-2L\eta)m}=\dfrac{2L\eta}{1-2L\eta}=\dfrac{1}{4}$. The condition number for $\grad^2 P$ is $\kp=\dfrac{L}{\gm}$, and usually $\kp\approx n$ (or $\kp=n$ for simplicity) where $n$ is the sample size. Then we pick $\eta = \dfrac{0.1}{L}$ which gives 
$\dfrac{2L\eta}{1-2L\eta}=\dfrac{0.2}{0.8}=\dfrac{1}{4} \ , \ \dfrac{1}{\gm \eta (1-2L\eta)m}=\dfrac{1}{\gm \eta \cdot 0.8m}=\dfrac{n}{0.8m}$.
Take $m=50n$ will yield $\dfrac{n}{0.8m}=\dfrac{1}{4}$. 
\end{remark}

\begin{proof}
Let the i.i.d sequence $
i_t\in \{1,\ldots,n\}$, and
$\Prob(i_t=k)=\dfrac{1}{n}$. Then we write the SVRG-update as 
$$w^{(t)} = w^{(t-1)}-\eta V_t$$
where
\[
V_t
=
\nabla \psi_{i_t}(w^{(t-1)})
-
\nabla \psi_{i_t}(\widetilde w)
+
\mathbb{E}\,\nabla \psi_{i_t}(\widetilde w) \ ,
\]
where
$\mathbb{E}(\bullet):=\E(\bullet\mid \mathcal{F}_{t-1})$. We expand
$$\begin{array}{ll}
\mathbb{E} \|w^{(t)}-w^*\|^2
& =
\mathbb{E} \|w^{(t-1)}-\eta V_t-w^*\|^2
\\
& =
\|w^{(t-1)}-w^*\|^2
-
2\eta (w^{(t-1)}-w^*)^{T}\mathbb{E}V_t
+
\eta^2 \mathbb{E}\|V_t\|^2 \ .
\end{array}
$$
Then we calculate

$$\begin{array}{ll}
\mathbb{E}V_t
& =
\mathbb{E}\,\nabla \psi_{i_t}(w^{(t-1)})
-
\mathbb{E}\,\nabla \psi_{i_t}(\widetilde w)
+
\mathbb{E}\,\nabla \psi_{i_t}(\widetilde w)
\\
& =
\dfrac{1}{n}\sum\li_{i=1}^{n}\nabla \psi_i(w^{(t-1)})
\\
& =
\nabla P(w^{(t-1)}) \ .
\end{array}
$$
And
$$\begin{array}{ll}
\mathbb{E}\|V_t\|^2
& = \mathbb{E}\left\| \underbrace{\nabla {\psi_{i_t}(w^{(t-1)})-\nabla \psi_{i_t}(w^*)}}_{(I)}
-
\underbrace{\Big[
\nabla \psi_{i_t}(\widetilde w)-\nabla \psi_{i_t}(w^*)-\nabla P(\widetilde w)}_{(II)}
\Big]\right\|^2
\\
& = \mathbb{E}\|(I)-(II)\|^2
\\
& \leq 2\mathbb{E}\|(I)\|^2 + 2\mathbb{E}\|(II)\|^2
\\
& \stackrel{(a)}{=}
2\,\mathbb{E}\|\nabla \psi_{i_t}(w^{(t-1)})-\nabla \psi_{i_t}(w^*)\|^2
+ 2\,\mathbb{E}
\Big\|
\big[\nabla \psi_{i_t}(\widetilde w)-\nabla \psi_{i_t}(w^*)\big]
-
\mathbb{E}\big[\nabla \psi_{i_t}(\widetilde w)-\nabla \psi_{i_t}(w^*)\big]
\Big\|^2
\\
&\stackrel{(b)}{\leq}
2\,\mathbb{E}\|\nabla \psi_{i_t}(w^{(t-1)})-\nabla \psi_{i_t}(w^*)\|^2 +
2\,\mathbb{E}\|\nabla \psi_{i_t}(\widetilde w)-\nabla \psi_{i_t}(w^*)\|^2 \ .
\end{array}
$$
Here in (a) we used the fact that $\nabla P(\widetilde w)=\mathbb{E}{\nabla \psi_{i_t}(\widetilde w)}$ and $\nabla P(w^*)=\mathbb{E}\grad \psi_{i_t}(w^*)=0$, and in (b) we used the fact that $
\mathbb{E}\|x-\mathbb{E}x\|^2\le \mathbb{E}\|x\|^2$. So we have in conclusion 
\begin{equation}\label{Thm:ConvexFunctionMainTheorem:Eq:IterationEstimate}
\begin{array}{ll}
\mathbb{E}\|w^{(t)}-w^*\|^2
& \leq \|w^{(t-1)}-w^*\|^2 -
2\eta (w^{(t-1)}-w^*)^{T}\nabla P(w^{(t-1)})
\\
& \qquad +2\eta^2\left(\mathbb{E}\|\nabla \psi_{i_t}(w^{(t-1)})-\nabla \psi_{i_t}(w^*)\|^2 
+
\mathbb{E}\|\nabla \psi_{i_t}(\widetilde w)-\nabla \psi_{i_t}(w^*)\|^2\right) \ . 
\end{array}
\end{equation}
By (\ref{Thm:GeneralizedStrongConvexity:Eq:UpperBound}) in Theorem \ref{Thm:GeneralizedStrongConvexity} we get
$$\|\nabla \psi_i(w)-\nabla \psi_i(w^*)\|^2
\le
2L\Big[
\psi_i(w)-\psi_i(w^*)
-
\nabla \psi_i(w^*)^{T}(w-w^*)
\Big] \ .$$
So
$$\begin{array}{ll}
& 2\,\mathbb{E}\|\nabla \psi_{i_t}(w^{(t-1)})-\nabla \psi_{i_t}(w^*)\|^2
+
2\,\mathbb{E}\|\nabla \psi_{i_t}(\widetilde w)-\nabla \psi_{i_t}(w^*)\|^2
\\
& \leq 2\cdot 2L\,
\mathbb{E}
\Big[
\psi_{i_t}(w^{(t-1)})-\psi_{i_t}(w^*)
-
\nabla \psi_{i_t}(w^*)^{T}(w^{(t-1)}-w^*)
\Big]
\\
& \qquad 
+ 2\cdot 2L\,
\mathbb{E}
\Big[
\psi_{i_t}(\widetilde w)-\psi_{i_t}(w^*)
-
\nabla \psi_{i_t}(w^*)^{T}(\widetilde w-w^*)
\Big]
\\
& =
4L
\Big[
P(w^{(t-1)})-P(w^*)-\nabla P(w^*)^{T}(w^{(t-1)}-w^*)
\Big]
+ 4L
\Big[
P(\widetilde w)-P(w^*)-\nabla P(w^*)^{T}(\widetilde w-w^*)
\Big]
\\
& = 4L\Big[P(w^{(t-1)})-P(w^*)+P(\widetilde w)-P(w^*)\Big] \ .
\end{array}
$$

This combined with (\ref{Thm:ConvexFunctionMainTheorem:Eq:IterationEstimate}) gives 
$$
\mathbb{E}\|w^{(t)}-w^*\|^2
\leq
\|w^{(t-1)}-w^*\|^2
-
2\eta (w^{(t-1)}-w^*)^{T}\nabla P(w^{(t-1)})
+
4L\eta^2\Big[P(w^{(t-1)})-P(w^*)+P(\widetilde w)-P(w^*)\Big] \ .
$$
Since $P(w)$ is convex, we get $P(w^{(t-1)})+(w^*-w^{(t-1)})^{T}\nabla P(w^{(t-1)})\le P(w^*)$. So we can bound $-2\eta (w^{(t-1)}-w^*)^{T}\nabla P(w^{(t-1)})\leq
2\eta [P(w^*)-P(w^{(t-1)})]$. This gives
$$\begin{array}{ll}
\mathbb{E}\|w^{(t)}-w^*\|^2 & \le
\|w^{(t-1)}-w^*\|^2
+
2\eta [P(w^*)-P(w^{(t-1)})]
\\
& \qquad +
4L\eta^2 [P(w^{(t-1)})-P(w^*)+P(\tilde w)-P(w^*)]
\\
& =
\|w^{(t-1)}-w^*\|^2
-
2\eta (1-2L\eta)\,[P(w^{(t-1)})-P(w^*)]
+
4L\eta^2 [P(\widetilde w)-P(w^*)]
\end{array}
$$
for $t=1,2,...$. Summing in one epoch, starting from $w^{(0)}=\widetilde{w}=\widetilde{w}_{s-1}$ and running from $t=1$ to $t=m$ gives 
$$\sum\li_{t=1}^m \mathbb{E}\|w^{(t)}-w^*\|^2\leq \sum\li_{t=1}^m \|w^{(t-1)}-w^*\|^2-2\eta(1-2L\eta)\sum\li_{t=1}^m [P(w^{(t-1)})-P(w^*)]+4L\eta^2m [P(\widetilde{w})-P(w^*)] \ .$$
Take $\E$ expectation on both sides and cancel $\E\|w^{(m)}-w^*\|$ for $t=2,...,m-1$ we get
$$\E\|w^{(m)}-w^*\|^2-\E\|\widetilde{w}_{s-1}-w^*\|^2\leq -2\eta(1-2L\eta)\sum\li_{t=1}^m [\E P(w^{(t-1)})-P(w^*)]+4L\eta^2m [\E P(\widetilde{w}_{s-1})-P(w^*)] \ .$$
Since $\widetilde{w}_s$ is randomly uniformly taken from $w^{(0)},...,w^{(m-1)}$ we get $\sum\li_{t=1}^m \E P(w^{(t-1)})=m\E P(\widetilde{w}_s)$. So 
\begin{equation}\label{Thm:ConvergenceSVRG:FinalBoundStep1}
\E\|w^{(m)}-w^*\|^2-\E\|\widetilde{w}_{s-1}-w^*\|^2\leq -2\eta(1-2L\eta)m[\E P(\widetilde{w}_s)-P(w^*)]+4L\eta^2m [\E P(\widetilde{w}_{s-1})-P(w^*)] \ .
\end{equation}
Since $P(w)$ is $\gamma$-strongly convex, by (\ref{Thm:ConvexFunctionMainTheorem:Eq:SecondOrderLowerBound}) we have
$$
P(\widetilde{w}_{s-1})-P(w^*)-\frac{1}{2}\gamma \|\widetilde{w}_{s-1}-w^*\|^2
\ge
(\nabla P(w^*))^{T}(\widetilde{w}_{s-1}-w^*)=0 \ .
$$
So \begin{equation}\label{Thm:ConvergenceSVRG:FinalBoundStep2}
-\E\|\widetilde{w}_{s-1}-w^*\|^2\geq -\dfrac{2}{\gm}[\E P(\widetilde{w}_{s-1})-P(w^*)] \ .
\end{equation}
Combining (\ref{Thm:ConvergenceSVRG:FinalBoundStep1}) and (\ref{Thm:ConvergenceSVRG:FinalBoundStep2}), using $\E\|w^{(m)}-w^*\|^2\geq 0$ we get
$$2\eta (1-2L\eta)m [\E P(\widetilde{w}_s)-P(w^*)]\leq \left(4L\eta^2 m +\dfrac{2}{\gm}\right)[\E P(\widetilde{w}_{s-1})-P(w^*)] \ ,$$
i.e.
$$\E P(\widetilde{w}_s)-P(w^*) \leq \left(\dfrac{2L\eta}{1-2L\eta} +\dfrac{1}{\gm\eta (1-2L\eta)m}\right)[\E P(\widetilde{w}_{s-1})-P(w^*)]\equiv \al [\E P(\widetilde{w}_{s-1})-P(w^*)] \ .$$
Iterating the above we get (\ref{Thm:ConvergenceSVRG:Eq:LinearConvergence}).
\end{proof}

\subsection{Variants of SVRG: SAG, SAGA, SARAH, SPIDER, STORM}

SVRG has various variants of the same variance-reduction nature and we introduce them below.

Stochastic Average Gradient (SAG, see \cite{SAGpaper}) has guaranteed linear convergence rate for strongly convex objective $P(x)=\dfrac{1}{n}\sum\li_{i=1}^n f_i(x)$. It is a variance-reduced optimization method that maintains and incrementally updates an average of gradients evaluated at previous iterates. At each iteration, it samples a single component function $f_i$, computes its gradient at the current point, and replaces the stored gradient for that index while keeping others unchanged. The algorithm then updates the iterate using the average of all stored gradients. In this way, SAG combines the low per-iteration cost of stochastic gradient methods with a progressively more accurate approximation of the full gradient, leading to faster (typically linear) convergence for strongly convex problems.

\begin{algorithm}[ht]
\caption{Stochastic Average Gradient (SAG): minimizing $P(x)=\dfrac{1}{n}\sum\li_{i=1}^n f_i(x)$}
\label{Alg:SAG}
\begin{algorithmic}[1]
\STATE \textbf{Input}: Input initialization $x_0$, stepsize $\al>0$, $d=0$, $y=0$, $i=1,2,...,n$
\FOR {$k=0,1,..., K-1$}
    \STATE Sample $i$ uniformly randomly from $\{1,2,...,n\}$
    \STATE $d \leftarrow d-y+\grad f_i(x)$
    \STATE $y=\grad f_i(x)$
    \STATE $x \leftarrow x-\dfrac{\al}{n}d$
\ENDFOR
\STATE \textbf{Output}: $x_K$
\end{algorithmic}
\end{algorithm}


SAGA (see \cite{SAGApaper}) has guaranteed linear convergence rate for strongly convex objective $P(x)=\dfrac{1}{n}\sum\li_{i=1}^n f_i(x)$. It is a variance-reduced stochastic optimization method that improves upon SAG by maintaining a table of past gradients and using a correction term to produce an unbiased gradient estimator at each iteration. Specifically, it updates one component gradient at a time and combines the new gradient with a correction based on the difference between the current and stored gradients, together with the average of all stored gradients. This construction ensures that the stochastic direction is an unbiased estimate of the full gradient while significantly reducing variance, leading to fast (often linear) convergence for strongly convex problems and improved theoretical guarantees compared to SAG.

\begin{algorithm}[ht]
\caption{SAGA: minimizing $P(x)=\dfrac{1}{n}\sum\li_{i=1}^n f_i(x)$}
\label{Alg:SAGA}
\begin{algorithmic}[1]
\STATE \textbf{Input}: Input initial vector $x^0\in \R^d$ and known gradients $\grad f_i(\phi_i^0)\in \R^d$ with $\phi_i^0=x^0$
                        for each $i$. These derivatives are stored in an $n\times d$ matrix.
\FOR {$k=0,1,..., K-1$}
    \STATE Sample $j$ uniformly randomly from $\{1,2,...,n\}$
    \STATE Take $\phi_j^{k+1}=x^k$ and store $\grad f_j(\phi_j^{k+1})$ in the table. 
            All other entries in the table remain unchanged. The quantity $\phi_j^{k+1}$ is not explicitly stored.
    \STATE Update $x$ using $\grad f_j(\phi_j^{k+1})$, $\grad f_j(\phi_j^k)$ and the table average:
            $$w^{k+1}=x^k-\gm \left[\grad f_j(\phi_j^{k+1})-\grad f_j(\phi_j^k)
                +\dfrac{1}{n}\sum\li_{i=1}^n \grad f_j(\phi_i^k)\right] \ ,$$
                $$x^{k+1}=\text{prox}_\gm^h (w^{k+1})=\text{arg}\min\li_{x\in \R^d}\left\{h(x)+\dfrac{1}{2\gm}\|x-w^{k+1}\|^2\right\} \ .$$
\ENDFOR
\STATE \textbf{Output}: $x^K$
\end{algorithmic}
\end{algorithm}


StochAstic Recursive grAdient algoritHm (SARAH, see \cite{SARAHpaper}) has guaranteed linear convergence rate for strongly convex objective $P(w)=\dfrac{1}{n}\sum\li_{i=1}^n f_i(w)$. It is a variance-reduced stochastic optimization method that constructs a recursive gradient estimator along the trajectory. Starting from a full gradient at the beginning of each outer loop, SARAH updates the estimator using differences of stochastic gradients between consecutive iterates, thereby maintaining a low-variance approximation of the true gradient without storing past gradients. This recursive structure allows the estimator variance to decrease as the iterates approach the optimum, leading to efficient convergence behavior comparable to SVRG-type methods, while often being more memory-efficient and practically stable.

\begin{algorithm}[ht!]
\caption{StochAstic Recursive grAdient algoritHm (SARAH): minimizing $P(w)=\dfrac{1}{n}\sum\li_{i=1}^n f_i(w)$}
\label{Alg:SARAH}
\begin{algorithmic}[1]
\STATE \textbf{Input}: Initialization $\widetilde{w}_0$, the learning rate $\eta>0$ and the inner loop size $m$
\FOR {$s=1,2,...$}
    \STATE $w_0=\widetilde{w}_{s-1}$
    \STATE $v_0=\dfrac{1}{n}\sum\li_{i=1}^n \grad f_i(w_0)$
    \STATE $w_1=w_0-\eta v_0$
    \FOR {$t=1,...,m-1$}
        \STATE Sample $i_t$ uniformly randomly in $\{1,2,...,n\}$
        \STATE $v_t=\grad f_{i_t}(w_t)-\grad f_{i_t}(w_{t-1})+v_{t-1}$
        \STATE $w_{t+1}=w_t-\eta v_t$
    \ENDFOR
    \STATE Set $\widetilde{w}_s=w_t$ with $t$ chosen uniformly randomly from $\{0,1,...,m-1\}$
\ENDFOR
\STATE \textbf{Output}: $\widetilde{w}_s$
\end{algorithmic}
\end{algorithm}


Stochastic Path--Integrated Differential EstiamtoR (SPIDER, see \cite{SPIDERpaper}) requires no assumption on convexity. Optimal Rate $O\left(\min(n^{1/2}\ve^{-2}, \ve^{-3})\right)$ for 
finding an $\ve$--approximate first order stationary point. It is a variance-reduced method designed for efficiently finding first-order stationary points, especially in nonconvex optimization. It constructs a recursive gradient estimator similar to SARAH, but periodically refreshes it using a large batch (or full gradient) and then updates it through small mini-batch differences along the optimization path. This path-integrated structure ensures that the estimator maintains high accuracy with significantly reduced computational cost. Combined with carefully chosen step sizes, SPIDER achieves near-optimal complexity for stochastic nonconvex optimization while controlling gradient noise effectively.

\begin{algorithm}[H]
\caption{Stochastic Path--Integrated Differential EstiamtoR (SPIDER): Searching First Order stationary point}
\label{Alg:SPIDER}
\begin{algorithmic}[1]
\STATE \textbf{Input}: Input $\boldx^0$, $q$, $S_1$, $S_2$, $n_0$, $\ve$ and $\widetilde{\ve}$
\FOR {$k=0,1,..., K-1$}
    \IF {$\text{mod}(k,q)=0$} 
        \STATE Draw $S_1$ samples (or compute the full gradient in the finite sum case), let $\boldv^k=\grad f_{\cS_1}(\boldx^k)$ 
    \ELSE
        \STATE Draw $S_2$ samples, and let $\boldv^k=\grad f_{\cS_2}(\boldx^k)-\grad f_{\cS_2}(\boldx^{k-1})+\boldv^{k-1}$
    \ENDIF
    \STATE \textbf{---Option I---}
        \IF {$\|\boldv^k\|\leq 2\widetilde{\ve}$}
            \STATE Return $\boldx^k$
        \ELSE 
            \STATE $\eta =\dfrac{\ve}{L n_0}$
            \STATE $\boldx^{k+1}=\boldx^k-\eta \dfrac{\boldv^k}{\|\boldv^k\|}$
        \ENDIF
    \STATE \textbf{---Option II---}
        \STATE $\eta^k =\min\left(\dfrac{\ve}{L n_0\|\boldv^k\|}, \dfrac{1}{2L n_0}\right)$
        \STATE $\boldx^{k+1}=\boldx^k-\eta^k \boldv^k$
\ENDFOR
\STATE \textbf{Output}: Output $\widetilde{\boldx}$ uniformly at random from $\{\boldx^k\}_{k=0}^{K-1}$
\end{algorithmic}
\end{algorithm}

\begin{remark}\label{Remark:Minibatch}
\rm In the classical SGD algorithm, the minibatches $\cB$ are sampled uniformly randomly from the index set $\{1,2,...,n\}$ with fixed minibatch size $m\leq n$. This kind of minibatches can be regarded as \emph{``sample without replacement"} minibatches. This is the commonly understood minibatches in practice.

In the SPIDER algorithm, the samples $\cS_1$ and $\cS_2$ are drawn with replacement, so that they form minibatches $\cS$ that are sampled uniformly randomly from the index set $\{1,2,...,n\}$ but with replacement, and the minibatch has a given size, say $m$ but counting multiplicities. This kind of minibacthes can be regarded as \emph{``sample with replacement"} minibacthes. This is less seen in the literature.

The reason why ``sample with replacement" minibatches are used in the SPIDER algorithm is because of the following fact: consider estimating the variance of the stochastic estimator $\dfrac{1}{m}\sum\li_{i\in \cS}\grad f_i(x)$ where $\cS$ is the ``sample with replacement" minibatch with size $m$, then
$$\E\left\|\dfrac{1}{m}\sum\li_{i\in \cS}\grad f_i(x)-\grad f(x)\right\|^2=
\E\left\|\dfrac{1}{m}\sum\li_{i\in \cS}(\grad f_i(x)-\grad f(x))\right\|^2=\dfrac{1}{m}\E\|\grad f_i(x)-\grad f(x)\|^2 \ ,$$
where in the last expectation the variable $i$ is taken uniformly randomly from $\{1,2,...,n\}$.

On the contrary, if $\cB$ is the ``sample without replacement" minibatch with the same size $m$, then instead we have
$$\begin{array}{ll}
\E\left\|\dfrac{1}{m}\sum\li_{i\in \cB}\grad f_i(x)-\grad f(x)\right\|^2& =
\E\left\|\dfrac{1}{m}\sum\li_{i\in \cB}(\grad f_i(x)-\grad f(x))\right\|^2 
\\
&\leq
\E\sum\li_{i\in \cB}\dfrac{1}{m^2}\sum\li_{i\in \cB}\|\grad f_i(x)-\grad f(x)\|^2 
=\E\|\grad f_i(x)- \grad f(x)\|^2 \ .
\end{array}$$
So we see a factor of $1/m$ is lost.
\end{remark}

Previous mentioned methods typically use \textit{non-adaptive
learning rates} and \textit{reliance on giant batch sizes} to construct variance reduced gradients through the use
of low-noise gradients calculated at a ``checkpoint".

STOchastic Recursive Momentum (STORM, see \cite{STORMpaper}) is a recently proposed novel variance-reduction method that achieves variance reduction through the use of a variant of the momentum term. 

This method never needs to compute a checkpoint gradient and achieves the optimal convergence rate of $O(1/T^{1/3})$. In fact, the key ideas in momentum-based methods 
is to replace the gradient estimator with an exponential moving avearge of the gradient estimators in all previous iteration steps, i.e.

$$
\begin{array}{ll}
\boldd_{t}&= (1-a)\boldd_{t-1}+a\grad f(\boldx_t, \xi_{t}) \ ,
\\
\boldx_t & =\boldx_{t-1}-\eta \boldd_t \ ,
\end{array}
$$
where $\grad f(x, \xi)$ is the usual stochastic gradient estimator. The STORM estimator incorporates ideas in SARAH estimator to modify the momentum updates as

$$
\begin{array}{ll}
\boldd_{t}&= (1-a)\boldd_{t-1}+a\grad f(\boldx_t, \xi_{t})+(1-a)(\grad f(\boldx_t, \xi_t)-\grad f(\boldx_{t-1}, \xi_t)) \ ,
\\
\boldx_t & =\boldx_{t-1}-\eta \boldd_t \ .
\end{array}
$$

We can think of the momentum iteration as an ``exponential moving average" version of SARAH, that is
$$\boldd_{t}= (1-a)\left[\boldd_{t-1}+(\grad f(\boldx_t, \xi_t)-\grad f(\boldx_{t-1}, \xi_t))\right]+a\grad f(\boldx_t, \xi_{t}) \ .
$$
Here the first part is like SARAH update, but it is weighted with the stochastic gradient update. Moreover, different from the SARAH update, the STORM update does not have to compute checkpoint gradients. Rather, variance-reduction is achieved by tuning the weight parameter $a$ appropriately. The exact algorithm goes as follows:

\begin{algorithm}[H]
	\caption{STOchastic Recursive Momentum (STORM)}
	\label{Alg:STORM}
	\begin{algorithmic}[1]
		\STATE \textbf{Input}: Parameters $k, w, c$, initial point $\boldx_1$
		\STATE Sample $\xi_1$
		\STATE $G_1\leftarrow \|\grad f(\boldx_1,\xi_1)\|$
		\STATE $\boldd_1\leftarrow \grad f(\boldx_1, \xi_1)$
		\STATE $\eta_0 \leftarrow k/w^{1/3}$
		\FOR {$t=1,2,...,T$}
			\STATE $\eta_t \leftarrow \dfrac{k}{(w+\sum_{i=1}^t G_i^2)^{1/3}}$
			\STATE $\boldx_{t+1}\leftarrow \boldx_t-\eta_t \boldd_t$
			\STATE $a_{t+1}\leftarrow c\eta_t^2$
			\STATE Sample $\xi_{t+1}$
			\STATE $G_{t+1}\leftarrow \|\grad f(\boldx_{t+1}, \xi_{t+1})\|$
			\STATE $\boldd_{t+1}\leftarrow(1-a_{t+1})\boldd_t+a_{t+1} \grad f(\boldx_{t+1}, \xi_{t+1})+(1-a_{t+1})[\grad f(\boldx_{t+1}, \xi_{t+1})-\grad f(\boldx_t, \xi_{t+1})]$
		\ENDFOR
		\STATE \textbf{Output}: $\widehat{\boldx}$ sampled uniformly randomly from $\boldx_1,..., \boldx_T$ (In practice, set $\widehat{\boldx}=\boldx_T$)
	\end{algorithmic}
\end{algorithm}


\section{Escape from saddle points and nonconvex problems, SGLD, Perturbed GD}

SGD and its variants can be viewed from continuous point as Stochastic Differential Equations (SDEs), see \cite{HuLiLiLiuDiffusionApproximateSGD}.
This inspires SGLD (see \cite{SGLDpaper}), which is a continuous version of SGD with additional moderate and isotropic noise. Its convergence is guaranteed at \cite{SGLDConvergence}. 

\begin{algorithm}[H]
\caption{Stochastic Gradient Langevin Dynamics: minimizing $P(w)=\dfrac{1}{n}\sum\li_{i=1}^n f_i(w)$}
\label{Alg:SGLD}
\begin{algorithmic}[1]
\STATE \textbf{Input}: Input initialization $w_0\in \R^d$, stepsize $\al>0$, inverse temperature $\bt>0$
\FOR {$k=0,1,..., K-1$}
    \STATE Sample a random minibatch $B\subset \{1,2,...,n\}$
    \STATE Calculate $\grad f_i(w_k)$ for $i\in B$
    \STATE Sample i.i.d. $\ve_k\sim \cN(0, I_d)$
    \STATE $w_{k+1}=w_k-\dfrac{\eta}{|B|} \sum\li_{i\in B}\grad f_i(w_k)+\sqrt{2\eta \bt^{-1}}\ve_k$
\ENDFOR
\STATE \textbf{Output}: $w_K$
\end{algorithmic}
\end{algorithm}


Perturbed GD (see \cite{PerturbedGDpaper}) is another variant inspired by continuous SGD, and it has guaranteed escape rate from saddle points.


\begin{algorithm}[H]
\caption{Perturbed Gradient Descent: PGD$(\boldx_0, \ell, \rho, \ve, c, \dt, \Dt_f)$}
\label{Alg:PGD}
\begin{algorithmic}[1]
\STATE \textbf{Input}: Input Initialization $\boldx_0$. 
\STATE Parameters $\chi \leftarrow 3\max\left\{\ln\left(\frac{d\ell \Dt_f}{c\ve^2\dt}\right), 4\right\}$,
$\eta \leftarrow \frac{c}{\ell}$, $r \leftarrow \frac{\sqrt{c}}{\chi^2}\cdot \frac{\ve}{\ell}$, 
$g_{\text{thres}}\leftarrow \frac{\sqrt{c}}{\chi^2}\cdot \ve$, 
$f_{\text{thres}}\leftarrow \frac{c}{\chi^3}\cdot \sqrt{\frac{c^3}{\rho}}$,
$t_{\text{thres}}\leftarrow \frac{\chi}{c^2}\cdot \frac{\ell}{\sqrt{\rho\ve}}$,
$t_{\text{noise}}\leftarrow -t_{\text{thres}}-1$. 
\FOR {$t=0,1,..., K-1$}
    \IF {$\|\grad f(\boldx_t)\|\leq g_{\text{thres}}$ and $t-t_{\text{noise}}>t_{\text{thres}}$}
        \STATE $\widetilde{\boldx}_t\leftarrow \boldx_t$, $t_{\text{noise}}\leftarrow t$
        \STATE $\boldx_t\leftarrow \widetilde{\boldx}_t+\xi_t$, $\xi_t$ uniformly $\sim \mathbb{B}_0(r)$
    \ENDIF
    \IF {$t-t_{\text{noise}}=t_{\text{thres}}$ and $f(\boldx_t)-f(\widetilde{\boldx}_{t_{\text{noise}}})>-f_{\text{thres}}$}
        \STATE \textbf{RETURN} $\widetilde{\boldx}_{t_{\text{noise}}}$
    \ENDIF
    \STATE $\boldx_{t+1}\leftarrow \boldx_t-\eta \grad f(\boldx_t)$
\ENDFOR
\STATE \textbf{Output}: $\boldx_K$
\end{algorithmic}
\end{algorithm}

\section{Coordinate Descent and Stochastic Coordinate Descent}

The $k$--th iteration of Coordinate Descent (see \cite{CoordinateDescentWrightPaper}, \cite{CoordinateDescentNesterovPaper}) applied to a function $f: \R^n\ra \R$ chooses some index $i_k\in \{1,2,...,n\}$
and takes iteration
$$x^{k+1}=x^k+\gm_k e_{i_k} \ ,$$
where $e_{i_k}$ is the $i_k$--th unit vector and $\gm_k$ is the step. 

\begin{example}
Gauss--Seidel method: 
$$\gm_k=\arg\min\li_{\gm}f(x^k+\gm e_{i_k})  \ .$$
\end{example}

\begin{example} Coordinate Gradient Descent method:
$$\gm_k=-\al_k \dfrac{\pt f}{\pt x_{i_k}}(x^k)e_{i_k}$$
for some $\al_k>0$. 
\end{example}

\begin{example}
If $i_k$ is chosen randomly the method is Stochastic Coordinate Descent.
\end{example}

Descent methods need $f(x^{k+1})<f(x^k)$ for all $k$.
